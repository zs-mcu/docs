## 时间格式修改

```js
下载库文件
yarn add moment



const moment = require('moment');
moment.locale('zh-cn')

module.exports = {
    themeConfig: {
        lastUpdated: '更新时间',
    },
    plugins: [
        [
            '@vuepress/last-updated',
            {
                transformer: (timestamp, lang) => {
                    //return moment(timestamp).fromNow()
                    return moment(timestamp).format("LLLL")
                }
            }
        ]
    ],
}
```
